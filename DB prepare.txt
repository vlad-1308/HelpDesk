CREATE DATABASE "HelpDesk"
    WITH
    OWNER = admin
    ENCODING = 'UTF8'
    LC_COLLATE = 'Russian_Russia.1251'
    LC_CTYPE = 'Russian_Russia.1251'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;




CREATE TABLE IF NOT EXISTS public.employees
(
    id integer NOT NULL,
    name character varying(30) COLLATE pg_catalog."default" NOT NULL,
    login character varying(50) COLLATE pg_catalog."default",
    password character varying(1000) COLLATE pg_catalog."default",
    active boolean,
    CONSTRAINT "Employees_pkey" PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.employees
    OWNER to admin;




CREATE TABLE IF NOT EXISTS public.entries
(
    id integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    theme character varying(100) COLLATE pg_catalog."default" NOT NULL,
    text character varying(5000) COLLATE pg_catalog."default",
    created_date date NOT NULL,
    created_by integer NOT NULL,
    modified_date date,
    modified_by integer,
    status character varying(20) COLLATE pg_catalog."default",
    CONSTRAINT "Entries_pkey" PRIMARY KEY (id),
    CONSTRAINT "Employees" FOREIGN KEY (created_by)
        REFERENCES public.employees (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT "Employees2" FOREIGN KEY (modified_by)
        REFERENCES public.employees (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.entries
    OWNER to admin;




CREATE TABLE IF NOT EXISTS public.user_role
(
    user_id integer NOT NULL,
    roles character varying(20) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT emp FOREIGN KEY (user_id)
        REFERENCES public.employees (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.user_role
    OWNER to admin;
